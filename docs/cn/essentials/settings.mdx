---

每个 Mintlify 站点都需要一个包含核心配置的 `docs.json` 文件。在下方了解更多[属性](#properties)。

## 属性

<ResponseField name="name" type="string" required>
项目名称。用作全局标题。

示例：`mintlify`

</ResponseField>

<ResponseField name="navigation" type="Navigation[]" required>
  包含所有分组及其页面的数组
  <Expandable title="Navigation">
    <ResponseField name="group" type="string">
    分组名称。

    示例：`Settings`

    </ResponseField>
    <ResponseField name="pages" type="string[]">
    作为页面的 Markdown 文件的相对路径。

    示例：`["customization", "page"]`

    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="logo" type="string or object">
  Logo 图片的路径，或包含 "light" 和 "dark" 模式 logo 图片路径的对象
  <Expandable title="Logo">
    <ResponseField name="light" type="string">
      浅色模式下 logo 的路径
    </ResponseField>
    <ResponseField name="dark" type="string">
      深色模式下 logo 的路径
    </ResponseField>
    <ResponseField name="href" type="string" default="/">
      点击 logo 后跳转的链接地址
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string">
  网站图标（favicon）的路径
</ResponseField>

<ResponseField name="colors" type="Colors">
  全局主题的十六进制颜色代码
  <Expandable title="Colors">
    <ResponseField name="primary" type="string" required>
      主色调。在浅色模式下，最常用于高亮内容、章节
      标题与点缀
    </ResponseField>
    <ResponseField name="light" type="string">
      深色模式下的主色调。最常用于高亮内容、章节标题与
      点缀（深色模式）
    </ResponseField>
    <ResponseField name="dark" type="string">
      用于重要按钮的主色调
    </ResponseField>
    <ResponseField name="background" type="object">
      浅色与深色模式下的背景颜色
      <Expandable title="Object">
        <ResponseField name="light" type="string" required>
          浅色模式下背景的十六进制颜色代码
        </ResponseField>
        <ResponseField name="dark" type="string" required>
          深色模式下背景的十六进制颜色代码
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarLinks" type="TopbarLink[]">
  要在顶栏中显示的链接的 `name` 与 `url` 数组
  <Expandable title="TopbarLink">
    <ResponseField name="name" type="string">
    按钮名称。

    示例：`Contact us`
    </ResponseField>
    <ResponseField name="url" type="string">
    点击按钮后跳转的 URL。示例：`https://mintlify.com/docs`
    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="topbarCtaButton" type="Call to Action">
  <Expandable title="Topbar Call to Action">
    <ResponseField name="type" type={'"link" or "github"'} default="link">
    Link 显示为一个按钮。GitHub 将显示给定 URL 处的仓库信息，包括 GitHub Star 数量。
    </ResponseField>
    <ResponseField name="url" type="string">
    若为 `link`：按钮链接的地址。
    
    若为 `github`：用于加载 GitHub 信息的仓库链接。
    </ResponseField>
    <ResponseField name="name" type="string">
    按钮内的文字。仅当 `type` 为 `link` 时必填。
    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="versions" type="string[]">
  版本名称数组。仅当您希望在导航栏中通过下拉菜单
  展示不同版本的文档时使用。
</ResponseField>

<ResponseField name="anchors" type="Anchor[]">
  锚点数组，包含 `icon`、`color` 与 `url`。
  <Expandable title="Anchor">
    <ResponseField name="icon" type="string">
    用于展示锚点的 [Font Awesome](https://fontawesome.com/search?q=heart) 图标。

    示例：`comments`
    </ResponseField>
    <ResponseField name="name" type="string">
    锚点标签的名称。

    示例：`Community`
    </ResponseField>
    <ResponseField name="url" type="string">
      用于标识哪些页面属于该锚点的 URL 起始部分。通常即您存放页面的文件夹名称。
    </ResponseField>
    <ResponseField name="color" type="string">
      锚点图标背景的十六进制颜色。也可以传入包含 `from` 与 `to`（均为十六进制颜色）属性的对象以使用渐变色。
    </ResponseField>
    <ResponseField name="version" type="string">
      用于在选中正确的文档版本之前隐藏锚点。
    </ResponseField>
    <ResponseField name="isDefaultHidden" type="boolean" default="false">
      若希望仅当通过直接链接进入其中的文档时才显示该锚点，请传入 `true`。
    </ResponseField>
    <ResponseField name="iconType" default="duotone" type="string">
      取值之一："brands"、"duotone"、"light"、"sharp-solid"、"solid" 或 "thin"
    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="topAnchor" type="Object">
  覆盖最顶部锚点的默认配置。
  <Expandable title="Object">
    <ResponseField name="name" default="Documentation" type="string">
      最顶部锚点的名称
    </ResponseField>
    <ResponseField name="icon" default="book-open" type="string">
      Font Awesome 图标。
    </ResponseField>
    <ResponseField name="iconType" default="duotone" type="string">
      取值之一："brands"、"duotone"、"light"、"sharp-solid"、"solid" 或 "thin"
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="tabs" type="Tabs[]">
  导航标签页数组。
  <Expandable title="Tabs">
    <ResponseField name="name" type="string">
      标签页标签的名称。
    </ResponseField>
    <ResponseField name="url" type="string">
      用于标识哪些页面属于该标签页的 URL 起始部分。通常即您存放
      页面的文件夹名称。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="api" type="API">
  API 设置的配置。在 [API 组件](/api-playground/demo)中了解更多关于 API 页面的信息。
  <Expandable title="API">
    <ResponseField name="baseUrl" type="string">
      所有 API 端点的基础 URL。如果 `baseUrl` 是一个数组，则会启用多个基础 URL
      选项，用户可在它们之间切换。
    </ResponseField>

    <ResponseField name="auth" type="Auth">
      <Expandable title="Auth">
        <ResponseField name="method" type='"bearer" | "basic" | "key"'>
          用于所有 API 端点的认证策略。
        </ResponseField>
        <ResponseField name="name" type="string">
        在 API 演练场中使用的认证参数名称。

        如果 method 为 `basic`，格式应为 `[usernameName]:[passwordName]`
        </ResponseField>
        <ResponseField name="inputPrefix" type="string">
        作为认证输入框前缀的默认值。

        例如，将 `inputPrefix` 设为 `AuthKey` 后，认证字段的默认输入结果会以 `AuthKey` 作为前缀。
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="playground" type="Playground">
      API 演练场的配置

      <Expandable title="Playground">
        <ResponseField name="mode" default="show" type='"show" | "simple" | "hide"'>
          演练场是显示、隐藏，还是仅展示端点而不提供用户交互（`simple`）

          在[演练场指南](/api-playground/demo)中了解更多
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="maintainOrder" type="boolean">
      启用该标志可确保 OpenAPI 页面中的键顺序与 OpenAPI 文件中定义的键顺序保持一致。

      <Warning>此行为很快将默认启用，届时该字段将被弃用。</Warning>
    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="openapi" type="string | string[]">
  指向您 OpenAPI 文件的 URL 或相对路径，可以是字符串或字符串数组。
  
  示例：
  <CodeGroup>
    ```json Absolute
    "openapi": "https://example.com/openapi.json"
    ```
    ```json Relative
    "openapi": "/openapi.json"
    ```
    ```json Multiple
    "openapi": ["https://example.com/openapi1.json", "/openapi2.json", "/openapi3.json"]
    ```
  </CodeGroup>

</ResponseField>

<ResponseField name="footerSocials" type="FooterSocials">
  社交媒体账户对象，键值对分别表示社交平台与账户 URL。
  
  示例：
  ```json
  {
    "x": "https://x.com/mintlify",
    "website": "https://mintlify.com"
  }
  ```
  <Expandable title="FooterSocials">
    <ResponseField name="[key]" type="string">
    以下取值之一：`website`、`facebook`、`x`、`discord`、`slack`、`github`、`linkedin`、`instagram`、`hacker-news`
    
    示例：`x`
    </ResponseField>
    <ResponseField name="property" type="string">
    指向社交平台的 URL。
    
    示例：`https://x.com/mintlify`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="feedback" type="Feedback">
  启用反馈按钮的配置

  <Expandable title="Feedback">
    <ResponseField name="suggestEdit" type="boolean" default="false">
    启用按钮，允许用户通过 pull request 建议修改
    </ResponseField>
    <ResponseField name="raiseIssue" type="boolean" default="false">
    启用按钮，允许用户对文档提交 issue
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="modeToggle" type="ModeToggle">
  自定义深色模式切换。
  <Expandable title="ModeToggle">
    <ResponseField name="default" type={'"light" or "dark"'}>
      如果希望对新用户始终展示浅色或深色模式，请进行设置。未设置时，
      默认与用户操作系统的模式一致。
    </ResponseField>
    <ResponseField name="isHidden" type="boolean" default="false">
      设为 true 可隐藏深色/浅色模式切换。您可以将 `isHidden` 与 `default` 结合使用，强制文档仅使用浅色或深色模式。例如：
      
      <CodeGroup>
      ```json Only Dark Mode
      "modeToggle": {
        "default": "dark",
        "isHidden": true
      }
      ```

      ```json Only Light Mode
      "modeToggle": {
        "default": "light",
        "isHidden": true
      }
      ```
      </CodeGroup>

    </ResponseField>

  </Expandable>
</ResponseField>

<ResponseField name="backgroundImage" type="string">
  在每个页面背后显示的背景图片。可参考
  [Infisical](https://infisical.com/docs) 与 [FRPC](https://frpc.io) 的示例。
</ResponseField>
